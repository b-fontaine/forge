# Flutter Architecture

## Clean Architecture + Feature-Sliced Design

Flutter projects use Clean Architecture organized by Feature-Sliced Design (FSD). Each feature is a vertical slice containing domain, data, and presentation layers. The dependency rule is absolute: inward only.

**Entity:**
```dart
// lib/features/auth/domain/entities/user.dart
import 'package:equatable/equatable.dart';

class User extends Equatable {
  final UserId id;
  final Email email;
  final UserName displayName;
  final UserStatus status;
  final DateTime createdAt;

  const User({
    required this.id,
    required this.email,
    required this.displayName,
    required this.status,
    required this.createdAt,
  });

  bool get isActive => status == UserStatus.active;

  User activate() {
    if (isActive) throw DomainException('User already active');
    return copyWith(status: UserStatus.active);
  }

  User copyWith({
    UserId? id,
    Email? email,
    UserName? displayName,
    UserStatus? status,
    DateTime? createdAt,
  }) => User(
    id: id ?? this.id,
    email: email ?? this.email,
    displayName: displayName ?? this.displayName,
    status: status ?? this.status,
    createdAt: createdAt ?? this.createdAt,
  );

  @override
  List<Object?> get props => [id];
}
```

**Mapper (model → domain entity):**
```dart
// lib/features/auth/data/mappers/user_mapper.dart
import 'package:injectable/injectable.dart';

@injectable
class UserMapper {
  User toDomain(UserModel model) => User(
    id: UserId(model.id),
    email: Email.fromTrusted(model.email), // pre-validated by API
    displayName: UserName(model.displayName),
    status: _mapStatus(model.status),
    createdAt: DateTime.parse(model.createdAt),
  );

  UserStatus _mapStatus(String raw) => switch (raw) {
    'active' => UserStatus.active,
    'inactive' => UserStatus.inactive,
    'suspended' => UserStatus.suspended,
    _ => throw DataException('Unknown user status: $raw'),
  };
}
```

**Repository Implementation:**
```dart
// lib/features/auth/data/repositories/auth_repository_impl.dart
import 'package:fpdart/fpdart.dart';
import 'package:injectable/injectable.dart';

@LazySingleton(as: AuthRepository)
class AuthRepositoryImpl implements AuthRepository {
  final AuthRemoteDataSource _remote;
  final AuthLocalDataSource _local;
  final UserMapper _mapper;
  final ErrorHandler _errorHandler;

  const AuthRepositoryImpl({
    required AuthRemoteDataSource remote,
    required AuthLocalDataSource local,
    required UserMapper mapper,
    required ErrorHandler errorHandler,
  })  : _remote = remote,
        _local = local,
        _mapper = mapper,
        _errorHandler = errorHandler;

  @override
  Future<Either<Failure, User>> signIn({
    required String email,
    required String password,
  }) async {
    try {
      final model = await _remote.signIn(email: email, password: password);
      await _local.cacheUser(model);
      return Right(_mapper.toDomain(model));
    } on DioException catch (e) {
      return Left(_errorHandler.fromDioException(e));
    } catch (e) {
      return Left(Failure.unexpected(e.toString()));
    }
  }

  @override
  Stream<User?> get authStateChanges => _remote.authStateChanges
      .map((model) => model != null ? _mapper.toDomain(model) : null);
}
```

### Router

```dart
// lib/core/router/app_router.dart
import 'package:go_router/go_router.dart';
import 'package:injectable/injectable.dart';

@singleton
class AppRouter {
  final AuthBloc _authBloc;

  AppRouter(this._authBloc);

  late final router = GoRouter(
    debugLogDiagnostics: false,
    redirect: _authGuard,
    routes: [
      GoRoute(
        path: '/',
        redirect: (_, __) => '/home',
      ),
      GoRoute(
        path: '/sign-in',
        builder: (_, __) => const SignInPage(),
      ),
      GoRoute(
        path: '/home',
        builder: (_, __) => const HomePage(),
        routes: [
          GoRoute(
            path: 'products/:id',
            builder: (context, state) => ProductDetailPage(
              productId: state.pathParameters['id']!,
            ),
          ),
        ],
      ),
    ],
  );

  String? _authGuard(BuildContext context, GoRouterState state) {
    final isAuthenticated = _authBloc.state is AuthStateAuthenticated;
    final isPublicRoute = ['/sign-in', '/sign-up'].contains(state.matchedLocation);

    if (!isAuthenticated && !isPublicRoute) return '/sign-in';
    if (isAuthenticated && isPublicRoute) return '/home';
    return null;
  }
}
```

## Feature Isolation Rules

1. **Features do not import from other features.** If feature B needs something from feature A, it belongs in `shared/` or `core/`.
2. **Cross-feature navigation uses the router, not constructors.** `context.go('/products/$id')` not `Navigator.push(ProductDetailPage())`.
3. **Shared domain concepts live in `shared/domain/`.** Email, PhoneNumber, Money, Pagination — used by multiple features.
4. **Shared widgets live in `shared/presentation/widgets/`.** AppButton, AppTextField, ErrorView.
5. **Feature BLoCs are provided at feature scope.** On the feature's page widget, not at the app level (unless truly global state).

```dart
// WRONG — feature-level BLoC at app level
void main() {
  runApp(
    MultiBlocProvider(
      providers: [
        BlocProvider(create: (_) => getIt<AuthBloc>()),
        BlocProvider(create: (_) => getIt<ProductBloc>()), // not needed globally
        BlocProvider(create: (_) => getIt<CartBloc>()),
      ],
      child: const App(),
    ),
  );
}

// RIGHT — scoped to where they're needed
class ProductListPage extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return BlocProvider(
      create: (_) => getIt<ProductBloc>()..add(const ProductsRequested()),
      child: const ProductListView(),
    );
  }
}
```

## Web: Lazy Loading with Deferred Imports

```dart
// lib/core/router/app_router.dart

// These modules are only downloaded when navigated to
import 'package:myapp/features/home/presentation/pages/home_page.dart'
    deferred as home;
import 'package:myapp/features/products/presentation/pages/products_page.dart'
    deferred as products;
import 'package:myapp/features/profile/presentation/pages/profile_page.dart'
    deferred as profile;

class _DeferredPage extends StatefulWidget {
  final Future<void> Function() loader;
  final Widget Function() child;

  const _DeferredPage({required this.loader, required this.child});

  @override
  State<_DeferredPage> createState() => _DeferredPageState();
}

class _DeferredPageState extends State<_DeferredPage> {
  late Future<void> _future;

  @override
  void initState() {
    super.initState();
    _future = widget.loader();
  }

  @override
  Widget build(BuildContext context) {
    return FutureBuilder(
      future: _future,
      builder: (context, snapshot) => switch (snapshot.connectionState) {
        ConnectionState.done when snapshot.hasError =>
            ErrorView(error: snapshot.error.toString()),
        ConnectionState.done =>
            widget.child(),
        _ =>
            const Scaffold(body: Center(child: CircularProgressIndicator())),
      },
    );
  }
}

final appRouter = GoRouter(
  routes: [
    GoRoute(
      path: '/home',
      builder: (_, __) => _DeferredPage(
        loader: home.loadLibrary,
        child: home.HomePage.new,
      ),
    ),
    GoRoute(
      path: '/products',
      builder: (_, __) => _DeferredPage(
        loader: products.loadLibrary,
        child: products.ProductsPage.new,
      ),
    ),
    GoRoute(
      path: '/profile',
      builder: (_, __) => _DeferredPage(
        loader: profile.loadLibrary,
        child: profile.ProfilePage.new,
      ),
    ),
  ],
);
```
